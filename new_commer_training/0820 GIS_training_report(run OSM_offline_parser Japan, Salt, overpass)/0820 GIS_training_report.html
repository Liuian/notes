singapore water

[out:json][timeout:25];
// Fetch the area of Singapore
{{geocodeArea:Singapore}}->.searchArea;

// Gather results for water bodies within Singapore
(
  way[natural=water](area.searchArea);
  relation[natural=water](area.searchArea);
);

// Print results
out body;
>;
out skel qt;
